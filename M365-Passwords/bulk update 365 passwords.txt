Install-Module -Name AzureAD
Install-Module -Name MSOnline


Import-Module AzureAD
Import-Module MSOnline

Connect-AzureAD

Connect-MsolService


Import-Csv -Path "C:\script\bulkuserpasswordchange.csv" | ForEach-Object {
    $userPrincipalName = $_.UserPrincipalName
    $users = Get-MsolUser | Where-Object { $_.UserPrincipalName -eq $userPrincipalName }

    if ($users.Count -gt 0) {
        foreach ($user in $users) {
            # Set the user's password and disable the password change requirement
            Set-MsolUserPassword -UserPrincipalName $user.UserPrincipalName -NewPassword $_.Password -ForceChangePassword $false
        }
    } else {
        Write-Host "User not found for UserPrincipalName: $userPrincipalName"
    }
}
